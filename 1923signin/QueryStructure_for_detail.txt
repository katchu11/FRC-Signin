SELECT Date_Format(a.logtime, '%d/%m/%y') AS LogDate, a.name_id, Date_Format(a.logtime, '%k:%i') AS checkinTime,
       (SELECT Date_Format(b.logtime, '%k:%i') FROM attendancelog b
        WHERE Date_Format(b.logtime, '%d/%m/%y') = Date_Format(a.logtime, '%d/%m/%y') AND
        b.name_id = a.name_id AND
        b.check_in_out_flag = 0) AS checkoutTime
FROM attendancelog a
WHERE a.check_in_out_flag = 1
